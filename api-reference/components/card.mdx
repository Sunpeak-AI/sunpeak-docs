## Overview

`Card` is a production-ready card component included in the sunpeak Starter Kit. It provides a flexible interface for displaying rich content with optional images, headers, metadata, and up to two action buttons.

## Import

```tsx
import { Card } from './components/card';
```

## Basic Usage

```tsx
<Card
  image="/photo.jpg"
  imageAlt="Scenic view"
  header="Card Title"
  metadata="⭐ 4.5 • Austin, TX"
  button1={{ children: "Primary Action", isPrimary: true, onClick: () => {} }}
>
  Card description text goes here.
</Card>
```

## Props

<ResponseField name="children" type="React.ReactNode">
  The main content/description displayed in the card body. Limited to 2 lines with ellipsis overflow.
</ResponseField>

<ResponseField name="image" type="string">
  URL or path to the image displayed at the top of the card.
</ResponseField>

<ResponseField name="imageAlt" type="string">
  Alt text for the image for accessibility.
</ResponseField>

<ResponseField name="imageMaxWidth" type="number" default="400">
  Maximum width of the image in pixels.
</ResponseField>

<ResponseField name="imageMaxHeight" type="number" default="400">
  Maximum height of the image in pixels.
</ResponseField>

<ResponseField name="header" type="React.ReactNode">
  The card header/title. Displays with medium font weight and truncates with ellipsis if too long.
</ResponseField>

<ResponseField name="metadata" type="React.ReactNode">
  Secondary metadata displayed below the header (e.g., ratings, location, price).
</ResponseField>

<ResponseField name="button1" type="OpenAIButtonProps">
  Configuration for the first action button.

  ```tsx
  interface OpenAIButtonProps {
    isPrimary?: boolean;
    onClick: () => void;
    children: React.ReactNode;
  }
  ```
</ResponseField>

<ResponseField name="button2" type="OpenAIButtonProps">
  Configuration for the second action button.
</ResponseField>

<ResponseField name="variant" type="'default' | 'bordered' | 'elevated'" default="default">
  Visual style variant of the card:
  - `default`: Subtle border with surface background.
  - `bordered`: Thicker border with surface background.
  - `elevated`: Subtle border with surface background and shadow.
</ResponseField>

<ResponseField name="className" type="string">
  Additional CSS classes to apply to the card container.
</ResponseField>

<ResponseField name="onClick" type="(e: React.MouseEvent<HTMLDivElement>) => void">
  Click handler for the entire card. Button clicks are prevented from bubbling to this handler.
</ResponseField>
