## Overview

The `ChatGPTSimulator` component provides a local development environment that replicates ChatGPT's widget runtime.

## Import

```tsx
import 'sunpeak/style.css'; // Required to style the simulator.
import { ChatGPTSimulator } from 'sunpeak';
```

## Props

<ResponseField name="children" type="ReactNode" required>
  Your widget component(s) to render in the simulator
</ResponseField>

<ResponseField name="initialTheme" type="'light' | 'dark'" default="'light'">
  Starting theme for the simulator
</ResponseField>

<ResponseField name="initialDisplayMode" type="'mobile' | 'tablet' | 'desktop'" default="'desktop'">
  Starting display mode
</ResponseField>

<ResponseField name="showControls" type="boolean" default="true">
  Show theme and display mode controls
</ResponseField>

<ResponseField name="className" type="string">
  Additional CSS classes
</ResponseField>

<ResponseField name="style" type="CSSProperties">
  Inline styles
</ResponseField>

## Usage

### Basic Usage

```tsx
import 'sunpeak/style.css'; // Required to style the simulator.
import { ChatGPTSimulator } from 'sunpeak';
import { MyWidget } from './MyWidget';

function App() {
  return (
    <ChatGPTSimulator>
      <MyWidget />
    </ChatGPTSimulator>
  );
}
```

### With Configuration

```tsx
<ChatGPTSimulator
  initialTheme="dark"
  initialDisplayMode="mobile"
  showControls={true}
>
  <MyWidget />
</ChatGPTSimulator>
```

### Multiple Widgets

```tsx
<ChatGPTSimulator>
  <WidgetOne />
  <WidgetTwo />
  <WidgetThree />
</ChatGPTSimulator>
```

### Custom Styling

```tsx
<ChatGPTSimulator
  className="custom-simulator"
  style={{ maxWidth: '1200px', margin: '0 auto' }}
>
  <MyWidget />
</ChatGPTSimulator>
```

## Best Practices

<AccordionGroup>
  <Accordion title="Development Only">
    Use the simulator for development only. In production, export your widget component directly.
  </Accordion>

  <Accordion title="Test All Modes">
    Test your widget in all display modes (mobile, tablet, desktop) and both themes (light, dark).
  </Accordion>

  <Accordion title="Keep Controls Visible">
    Leave `showControls={true}` during development to easily switch themes and display modes.
  </Accordion>

  <Accordion title="Use with HMR">
    The simulator works best with Vite's Hot Module Replacement for instant updates.
  </Accordion>
</AccordionGroup>
