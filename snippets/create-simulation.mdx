First, create a resource JSON file with metadata:

```json
// src/resources/example-resource.json
{
  "name": "example",
  "title": "Example",
  "description": "My example component",
  "mimeType": "text/html+skybridge",
  "_meta": {
    "openai/widgetDomain": "example.com",
    "openai/widgetCSP": {
      "connect_domains": [],
      "resource_domains": []
    }
  }
}
```

Then, create the simulation that imports the resource metadata:

```ts
// src/simulations/example-simulation.ts
import exampleResourceMeta from '../resources/example-resource.json';

export const exampleSimulation = {
  userMessage: 'Show me an example', // Stylistic, no functional purpose.

  // MCP Tool protocol.
  tool: {
    name: 'show-example',
    description: 'Show an example simulation',
    inputSchema: { type: 'object', properties: {}, additionalProperties: false } as const,
    title: 'Show Example',
    annotations: { readOnlyHint: true },
    _meta: {
      'openai/toolInvocation/invoking': 'Loading example',
      'openai/toolInvocation/invoked': 'Example loaded',
      'openai/widgetAccessible': true,
      'openai/resultCanProduceWidget': true,
    },
  },

  // MCP Resource protocol - imported from resource meta file.
  resource: exampleResourceMeta,

  // MCP CallTool protocol.
  toolCall: {
    structuredContent: null, // Inject dummy data here.
    _meta: {},
  },
} as const;
```
